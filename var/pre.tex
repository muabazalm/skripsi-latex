\def\@makechapterhead#1{%
  \begin{center}
  %%jarak tulisan BAB
  \vspace*{-50\p@}%jarak bab ke margin
  % {\reset@font\Large\bfseries% Besarin huruf chapter
  {\bfseries%
    \MakeUppercase{\@chapapp\space \thechapter}\\ [-.5em]%

  }
  \end{center}
}
